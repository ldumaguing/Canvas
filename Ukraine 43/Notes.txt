
update ChitStatus A
inner join chits B on A.aaa = B.id
set A.front = B.name;



update ChitStatus A
inner join chits B
on
A.bbb = B.id
set A.back = B.name;



update ChitStatus A
inner join chits B
on
A.ccc = B.id
set A.rem = B.name;





select 
*
from
 ChitStatus a,
 chits b
where
 a.rem = b.id;


create table ChitStatus (
   id int primary key,
   name varchar(80),
   front int,
   back int default null,
   rem int

);







create table chits (
   id int primary key,
   name varchar(80)
);
